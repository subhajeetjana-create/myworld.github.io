<script>

let noCount = 0;

const noBtn = document.getElementById("no");

const yesBtn = document.getElementById("yes");

const msg = document.getElementById("msg");

const card = document.getElementById("card");

const noStages = [

  { text: "😢 Oh no… that hurts.", bg: "#cfd9df", emoji: "💔" },

  { text: "🥺 Are you really sure?", bg: "#b6c1d4", emoji: "😔" },

  { text: "😭 I’m getting emotional now.", bg: "#a8b6cc", emoji: "🥀" },

  { text: "😂 Okay but NO is not an option.", bg: "#ffb3c1", emoji: "🙈" },

  { text: "😏 Button is scared now.", bg: "#ff758f", emoji: "😜" }

];

function createEmoji(char) {

  const e = document.createElement("div");

  e.className = "emoji";

  e.innerText = char;

  e.style.left = Math.random() * 100 + "vw";

  e.style.bottom = "0px";

  document.body.appendChild(e);

  setTimeout(() => e.remove(), 3000);

}

noBtn.onclick = () => {

  if (noCount < noStages.length) {

    msg.innerText = noStages[noCount].text;

    document.body.style.background = noStages[noCount].bg;

    for (let i = 0; i < 6; i++) {

      createEmoji(noStages[noCount].emoji);

    }

    noCount++;

  }

  if (noCount === noStages.length) {

    noBtn.style.position = "absolute";

    noBtn.onmouseenter = () => {

      noBtn.style.left = Math.random() * 70 + "%";

      noBtn.style.top = Math.random() * 60 + "%";

    };

  }

};

yesBtn.onclick = () => {

  document.body.style.background =

    "linear-gradient(135deg, #ff758f, #ff7eb3)";

  card.innerHTML = `

    <h2>😍💖</h2>

    <p style="font-size:20px;"><b>I LOVE YOU BABE ❤️</b></p>

    <p>Every day with you is my favourite journey 🥹</p>

    <button onclick="openVideo()">▶ Watch Our Memories</button>

  `;

  setInterval(() => {

    createEmoji("❤️");

    createEmoji("😊");

    createEmoji("🥰");

  }, 400);

};

function openVideo() {

  window.open(

    "https://drive.google.com/file/d/1Skb8E_dMvR6XyZDUVVt9mVrM5Ozrab08/view",

    "_blank"

  );

}
